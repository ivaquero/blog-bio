def plot_regression_predictions(tree_reg,
                                X,
                                y,
                                ax,
                                lims=[0, 1, -0.2, 1],
                                ylabel="$y$"):
    x1 = np.linspace(lims[0], lims[1], 500).reshape(-1, 1)
    y_pred = tree_reg.predict(x1)
    ax.set(xlabel="$x_1$")
    if ylabel:
        ax.set(ylabel=ylabel)
    ax.plot(X, y, "b.")
    ax.plot(x1, y_pred, "r.-", lw=2, label=r"$\hat{y}$")
